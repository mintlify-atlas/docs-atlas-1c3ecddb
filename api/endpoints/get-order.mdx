---
title: "Get Order"
api: "GET /api/orders/:id"
description: "Retrieve detailed information about a specific order"
---

## Authentication

Requires a valid API key in the `Authorization` header.

## Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the order to retrieve
</ParamField>

## Response

Returns a single order object with complete details including customer information, line items, delivery details, and order status.

<ResponseField name="id" type="string">
  Unique order identifier (UUID)
</ResponseField>

<ResponseField name="customer_name" type="string" nullable>
  Customer name extracted from the chat
</ResponseField>

<ResponseField name="items" type="array">
  Array of order line items
  <Expandable title="Item properties">
    <ResponseField name="product_name" type="string">
      Name of the product
    </ResponseField>
    <ResponseField name="quantity" type="number">
      Quantity ordered (must be positive)
    </ResponseField>
    <ResponseField name="price" type="number" nullable>
      Price per unit (can be null if not specified)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="delivery_address" type="string" nullable>
  Delivery address for the order
</ResponseField>

<ResponseField name="delivery_date" type="string" nullable>
  Requested delivery date (ISO 8601 format)
</ResponseField>

<ResponseField name="special_instructions" type="string" nullable>
  Special instructions or notes from the customer
</ResponseField>

<ResponseField name="total" type="number" nullable>
  Total order amount (numeric)
</ResponseField>

<ResponseField name="confidence" type="string">
  AI extraction confidence level: `high`, `medium`, or `low`
</ResponseField>

<ResponseField name="status" type="string" default="pending">
  Order status: `pending`, `confirmed`, `fulfilled`, or `cancelled`
</ResponseField>

<ResponseField name="created_at" type="string">
  Order creation timestamp (ISO 8601 format)
</ResponseField>

<ResponseField name="raw_messages" type="array">
  Original chat messages that were used to extract this order
  <Expandable title="Message properties">
    <ResponseField name="sender" type="string">
      Message sender identifier
    </ResponseField>
    <ResponseField name="text" type="string">
      Message content
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="invoice" type="object" nullable>
  Generated invoice details (if invoice has been generated)
  <Expandable title="Invoice properties">
    <ResponseField name="invoice_number" type="string">
      Unique invoice number
    </ResponseField>
    <ResponseField name="date" type="string">
      Invoice date
    </ResponseField>
    <ResponseField name="customer_name" type="string">
      Customer name on invoice
    </ResponseField>
    <ResponseField name="items" type="array">
      Invoice line items with pricing
    </ResponseField>
    <ResponseField name="subtotal" type="number">
      Subtotal before tax
    </ResponseField>
    <ResponseField name="cgst" type="number">
      Central GST amount
    </ResponseField>
    <ResponseField name="sgst" type="number">
      State GST amount
    </ResponseField>
    <ResponseField name="igst" type="number">
      Integrated GST amount (for interstate transactions)
    </ResponseField>
    <ResponseField name="total" type="number">
      Total invoice amount including taxes
    </ResponseField>
    <ResponseField name="business_name" type="string">
      Your business name
    </ResponseField>
    <ResponseField name="gst_number" type="string">
      Your GST registration number
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.chat2cash.ai/api/orders/ord_7f8a9b2c3d4e5f6g7h8i9j0k" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```javascript Node.js
const orderId = 'ord_7f8a9b2c3d4e5f6g7h8i9j0k';

const response = await fetch(`https://api.chat2cash.ai/api/orders/${orderId}`, {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY'
  }
});

const order = await response.json();
console.log(order);
```

```python Python
import requests

order_id = 'ord_7f8a9b2c3d4e5f6g7h8i9j0k'

response = requests.get(
    f'https://api.chat2cash.ai/api/orders/{order_id}',
    headers={'Authorization': 'Bearer YOUR_API_KEY'}
)

order = response.json()
print(order)
```
